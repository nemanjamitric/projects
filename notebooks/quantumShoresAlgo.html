from qiskit_aer import AerSimulator

backend = AerSimulator()

N = 15
maxAttempts = 30
successes = 0
aReselects = 0

for _ in range(maxAttempts):
    a = np.random.randint(2, N)
    try:
        job, circuit = shor.factor(N=N, a=a)
        result = shor.package_result(job, N, a)
        if result.factors:
            successes += 1
        aReselects += 1
    except:
        continue

maxN = 50
lastSuccess = None

for N in range(15, maxN, 2):

        job, circuit = shor.factor(N=N, a=2)
        result = shor.package_result(job, N, 2)
        if result.factors:
            print(f"Successfully factorized: {N}")
            lastSuccess = N
        else:
            break
    except:
        print(f"Error for N={N}")
        break

Največje število zaganjanj algoritma (100 testov)¶

Merimo maksimalno število zaganjanj kvantnega dela algoritma potrebnih za uspešno faktorizacijo (pri 100 naključnih primerih).

In [5]:
maxRetries = []
N = 15

for _ in tqdm.tqdm(range(100)):
    retries = 0
    success = False
    while retries < 50:
        a = np.random.randint(2, N)
        try:
            job, circuit = shor.factor(N=N, a=a)
            result = shor.package_result(job, N, a)
            if result.factors:
                success = True
                break
        except:
            pass
        retries += 1
    maxRetries.append(retries if success else 50)

print(f"Maximum attempts needed for success: {max(maxRetries)}")
print(f"Average number of attempts: {np.mean(maxRetries):.2f}")

import matplotlib.pyplot as plt
from qiskit import transpile

Ns = [15, 21, 33, 35, 39]
qubits = []
depths = []

for N in Ns:
    try:
        print(f"Building circuit for N = {N}")
        circ = shor.construct_circuit(N=N, a=2, measurement=True)
        circ = transpile(circ, backend)
        qubits.append(circ.num_qubits)
        depths.append(circ.depth())
    except Exception as e:
        print(f"Failed for N={N}: {e}")

plt.plot(qubits, depths, marker="o")
plt.xlabel("Število kubitov")
plt.ylabel("Globina vezja")
plt.title("Globina vezja glede na vhodno število")
plt.grid(True)
plt.show()

Na podlagi prejšnjih rezultatov ocenim, koliko časa bi potrebovala simulacija velikih števil.

logDepths = np.log10(depths)
fit = np.polyfit(qubits, logDepths, 1)
slope, intercept = fit

def predictLog10Time(qb):
    return slope * qb + intercept

def humanReadableRime(log10Seconds):
    seconds = 10 ** log10Seconds
    if seconds < 60:
        return f"{seconds:.2f} seconds"
    elif seconds < 3600:
        return f"{seconds / 60:.2f} minutes"
    elif seconds < 86400:
        return f"{seconds / 3600:.2f} hours"
    elif seconds < 31536000:
        return f"{seconds / 86400:.2f} days"
    else:
        return f"{seconds / 31536000:.2f} years"
