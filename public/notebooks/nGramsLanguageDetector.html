Language Profile Generator And Detector

Creates n-gram profiles for language detection using character trigrams or bigrams.

2. Profile Generation Functions

cleanText(text)

Remove all non-alphabetic characters except spaces, and normalize whitespace and lowercase

loadTrainingText(language, filePath)

Loads training text from PDF files for a language and converts it to lowercase.

  • Input: Language name (e.g., "slovenian"), directory path.
  • Output: Lowercase text as a single string.

extractNGrams(text, n)

Splits text into overlapping character n-grams (e.g., "hello" ["hel", "ell", "llo"]).

  • Input: Text string, n-gram length (default: 3).
  • Output: List of n-grams.

computeRankedNGrams(ngrams, topN=300)

Computes the top topN most frequent n-grams and ranks them (1 = most frequent).

  • Input: List of n-grams, number of top n-grams to keep.
  • Output: Dictionary {ngram: rank}.

saveProfile(profile, fileName)

Saves the n-gram profile as a JSON file.

def cleanAndSplit(text):
    text = re.sub(r'[^a-zA-Z-\s]', '', text)
    text = re.sub(r'\s+', ' ', text).strip()
    return text.lower()

def loadTrainingText(language, filePath):
    txt_path = f"{filePath}/{language}.txt"
    pdf_path = f"{filePath}/{language}.pdf"
    
    if os.path.exists(txt_path):
        with open(txt_path, "r", encoding="utf-8") as f:
            text = f.read()
    elif os.path.exists(pdf_path):
        try:
            with open(pdf_path, "rb") as f:
                reader = PyPDF2.PdfReader(f)
                text = "".join([page.extract_text() for page in reader.pages])
        except:
            text = extract_text(pdf_path)
    else:
        raise FileNotFoundError(f"No file found for {language}")
    
    return cleanAndSplit(text)

def extractNGrams(text, n):
    words = text.split()
    ngrams = []
    for word in words:
        if len(word) >= n:
            for i in range(len(word) - n + 1):
                ngrams.append(word[i:i+n])
    return ngrams

def computeRankedNGrams(ngrams, topN=300):
    counts = Counter(ngrams)
    ranked = sorted(counts.items(), key=lambda x: x[1], reverse=True)
    return {ngram: rank+1 for rank, (ngram, _) in enumerate(ranked[:topN])}

def saveProfile(profile, fileName):
    with open(fileName, "w", encoding="utf-8") as file:
        json.dump(profile, file)

    profile = computeRankedNGrams(ngrams)

def plotTopNgrams(lang, n=10):
    with open(f"{profilesDir}/{lang}Profile.json", "r", encoding="utf-8") as file:
        profile = json.load(file)
    topNgrams = sorted(profile.items(), key=lambda x: x[1])[:n]
    ngrams, ranks = zip(*topNgrams)
    
    plt.figure(figsize=(10, 4))
    plt.bar(ngrams, [1/rank for rank in ranks])
    plt.title(f"Top {n} N-Grams in {lang.capitalize()}")
    plt.xlabel("N-Gram")
    plt.ylabel("1 / Rank (higher = more frequent)")
    plt.xticks(rotation=45)
    plt.show()

for lang in languages:
    plotTopNgrams(lang)

def loadProfiles(profileDir):
    profiles = {}
    for filePath in glob.glob(f"{profileDir}/*.json"):
        lang = filePath.split("/")[-1].replace("Profile.json", "")
        with open(filePath, "r", encoding="utf-8") as file:
            profiles[lang] = json.load(file)
    return profiles

def detectPdfLanguage(pdfPath, profiles):
    try:
        with open(pdfPath, "rb") as f:
            reader = PyPDF2.PdfReader(f)
            text = "".join([page.extract_text() for page in reader.pages])
        cleanedText = cleanAndSplit(text)
        
        testNGrams = extractNGrams(cleanedText, defaultN)
        if not testNGrams:
            return "unknown"
        
        test_counts = Counter(testNGrams)
        testRanks = {ngram: rank+1 for rank, (ngram, _) in enumerate(test_counts.most_common())}
        
        minDistance = float('inf')
        predictedLang = "unknown"
        for lang, profile in profiles.items():
            distance = sum(
                abs(testRanks.get(ngram, 301) - profile.get(ngram, 301))
                for ngram in testRanks
            )
            if distance < minDistance:
                minDistance = distance
                predictedLang = lang
            # print(f"For file {pdfPath} to profile {lang}: {distance}")
        return predictedLang
    
    except Exception as e:
        print(f"Error processing {pdfPath}: {str(e)}")
        return "error"
