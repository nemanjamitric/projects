def LoadAudio(FilePath):
    AudioData, SampleRate = sf.read(FilePath)
    if SampleRate != SamplingRate:
        raise ValueError("Sampling rate mismatch")
    return AudioData[:3*SamplingRate]

def ComputeRealScalarProduct(Signal, Frequency, Length):
    TimeVector = np.arange(Length) / SamplingRate
    CosineBasis = np.cos(2 * np.pi * Frequency * TimeVector)
    SineBasis = np.sin(2 * np.pi * Frequency * TimeVector)
    CosineAmplitude = np.dot(Signal, CosineBasis) / Length
    SineAmplitude = np.dot(Signal, SineBasis) / Length
    return np.sqrt(CosineAmplitude**2 + SineAmplitude**2)

def ComputeComplexScalarProduct(Signal, Frequency, Length):
    TimeVector = np.arange(Length) / SamplingRate
    ComplexBasis = np.exp(-2j * np.pi * Frequency * TimeVector)
    ComplexAmplitude = np.dot(Signal, ComplexBasis) / Length
    return np.abs(ComplexAmplitude)

def CustomDFT(Signal, IsComplex=True):
    Frequencies = np.linspace(0, MaxFrequency, FrequencyBins)
    Spectrum = np.zeros(FrequencyBins)
    for Index, Freq in enumerate(Frequencies):
        if IsComplex:
            Spectrum[Index] = ComputeComplexScalarProduct(Signal, Freq, len(Signal))
        else:
            Spectrum[Index] = ComputeRealScalarProduct(Signal, Freq, len(Signal))
    return Frequencies, Spectrum

def ComputeFFT(Signal):
    FFTResult = np.fft.fft(Signal)
    Frequencies = np.fft.fftfreq(len(Signal), 1/SamplingRate)
    PositiveMask = (Frequencies >= 0) & (Frequencies <= MaxFrequency)
    return Frequencies[PositiveMask], np.abs(FFTResult[PositiveMask]) / len(Signal)

def PlotSpectrum(RealFrequencies, RealSpectrum, ComplexFrequencies, ComplexSpectrum, Title, FFTFrequencies=None, FFTSpectrum=None):
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 5), sharey=True)
    
    ax1.plot(RealFrequencies, RealSpectrum, label="Real DFT", color='blue')
    if FFTFrequencies is not None and FFTSpectrum is not None:
        ax1.plot(FFTFrequencies, FFTSpectrum, label="FFT", alpha=0.7, color='orange')
    ax1.set_title(f"{Title} - Real DFT")
    ax1.set_xlabel("Frequency (Hz)")
    ax1.set_ylabel("Amplitude")
    ax1.grid(True)
    ax1.legend()
    
    ax2.plot(ComplexFrequencies, ComplexSpectrum, label="Complex DFT", color='green')
    if FFTFrequencies is not None and FFTSpectrum is not None:
        ax2.plot(FFTFrequencies, FFTSpectrum, label="FFT", alpha=0.7, color='orange')
    ax2.set_title(f"{Title} - Complex DFT")
    ax2.set_xlabel("Frequency (Hz)")
    ax2.grid(True)
    ax2.legend()
    
    plt.tight_layout()
    plt.show()

def ComputeEnergyByFrequency(Frequencies, Spectrum):
    Energy = np.power(Spectrum, 2)
    TotalEnergy = np.sum(Energy)
    SignificantFrequencies = Frequencies[Energy > 0.01 * TotalEnergy]
    SignificantAmplitudes = Spectrum[Energy > 0.01 * TotalEnergy]
    return SignificantFrequencies, SignificantAmplitudes

def CompareDFTSpectra(RealFrequencies, RealSpectrum, ComplexFrequencies, ComplexSpectrum, Title="DFT Comparison"):
    if not np.array_equal(RealFrequencies, ComplexFrequencies):
        raise ValueError("Real and Complex DFT frequencies must match")
    
    AbsDifference = np.abs(RealSpectrum - ComplexSpectrum)
    
    RelativeDifference = np.zeros_like(AbsDifference)
    mask = ComplexSpectrum > 1e-10
    RelativeDifference[mask] = AbsDifference[mask] / ComplexSpectrum[mask]
    
    print(f"{Title} - statistika razlik:")
    print(f"Povprečna absolutna razlika (realno - kompleksno): {np.mean(AbsDifference):.4f}")
    print(f"Največja absolutna razlika (realno - kompleksno): {np.max(AbsDifference):.4f}")
    print(f"Povprečna relativna razlika (realno - kompleksno): {np.mean(RelativeDifference[mask]):.4f}")

for VowelKey, FilePath in VowelFiles.items():
    Signal = LoadAudio("./sounds/" + FilePath)
    
    FFTFrequencies, FFTSpectrum = ComputeFFT(Signal)
    RealFrequencies, RealSpectrum = CustomDFT(Signal, IsComplex=False)
    ComplexFrequencies, ComplexSpectrum = CustomDFT(Signal, IsComplex=True)

    SignificantFrequencies, SignificantAmplitudes = ComputeEnergyByFrequency(ComplexFrequencies, ComplexSpectrum)
    print(f"Pomembne frekvence samoglasnika {VowelKey} nizek (Hz):", SignificantFrequencies)
    print(f"Ustrezne amplitude:", SignificantAmplitudes)

Answers To Questions¶

1. Ali lahko frekvenčno sliko signala nedvoumno določimo preko skalarnega produkta med posnetim signalom in izbrano realno sinusoido? Zakaj?¶

Ne. Realni sinusoidi zajamejo le prispevke amplitude brez informacij o fazi. Več frekvenčnih komponent z različnimi fazami lahko proizvede enak rezultat skalarnih produktov. Grafi realne DFT (zgoraj) pogosto ne zajamejo podrobnosti, povezanih s fazo, v primerjavi z FFT, kar je razvidno iz nekoliko manj natančnih oblik vrhov.

2. Ali lahko frekvenčno sliko signala nedvoumno določimo preko skalarnega produkta med posnetim signalom in izbrano kompleksno sinusoido? Kako na to vpliva dolžina opazovanega intervala signala? Zakaj?¶

Da, frekvenčni spekter je mogoče enolično določiti z uporabo skalarnih produktov s kompleksnim sinusoidom. Kompleksni sinusoidi zajamejo tako amplitudo kot fazo. Dolžina intervala opazovanja vpliva na frekvenčno ločljivost: krajši intervali povzročijo širše vrhove, medtem ko daljši intervali izboljšajo ločljivost z oženjem vrhov. To je zato, ker je frekvenčna ločljivost obratno sorazmerna z dolžino signala (Δf = 1/T). Grafi kompleksne DFT se zelo ujemajo z FFT, kar potrjuje njihovo natančnost.

3. Katere vse frekvence so prisotne v izgovorjavi samoglasnikov »a«, »i« in »o« (osredotočite se na frekvence, ki predstavljajo vsaj 1 % celotne energije signala)?¶

Pomembne frekvence (ki prispevajo vsaj 1 % energije signala) so izpisane zgoraj za vsak samoglasnik.

4. Kako so frekvence, ki so prisotne v izgovorjavi samoglasnikov »a«, »i« in »o« odvisne od višine tona izgovorjave?¶

Višja višina poveča osnovno frekvenco in premakne harmonike na višje frekvence.

5. Kako se z višino tona izgovorjave spremeni pojav višjih harmonikov v frekvenčni sliki?¶

Višja višina poveča osnovno frekvenco, kar povzroči, da se harmoniki pojavijo pri višjih večkratnikih in pogosto z večjo amplitudo. To je razvidno iz grafov kompleksne DFT, kjer imajo visoki samoglasniki močnejše vrhove pri višjih frekvencah.
