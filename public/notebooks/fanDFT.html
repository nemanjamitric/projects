Funkcija DFT¶

In [3]:
def computeDft(signal, sampleRate, maxFreq=1000):
    sampleCount = len(signal)
    window = np.hanning(sampleCount)
    windowedSignal = signal * window
    fftResult = np.fft.fft(windowedSignal)
    mag = np.abs(fftResult[:sampleCount//2]) / sampleCount
    freqs = np.fft.fftfreq(sampleCount, d=1/sampleRate)[:sampleCount//2]
    mask = freqs <= maxFreq
    return freqs[mask], mag[mask]

def analyzeSignal(signalName, bladeCount, rpmRange, filePath):
    if not os.path.exists(filePath):
        print(f"File {filePath} not found")
        return None, None
    
    audio = AudioSegment.from_file(filePath, format="m4a")
    sampleRate = audio.frame_rate
    signal = np.array(audio.get_array_of_samples(), dtype=np.float32)
    if audio.channels > 1:
        signal = signal.reshape(-1, audio.channels).mean(axis=1)
    signal = signal / np.max(np.abs(signal))
    
    freqs, mag = computeDft(signal, sampleRate)
    fundamentalFreq = freqs[np.argmax(mag)]
    estimatedRpm = (fundamentalFreq * 60) / bladeCount
    rpmStatus = "within range" if rpmRange[0] <= estimatedRpm <= rpmRange[1] else "outside range"
    print(f"{signalName}: Fundamental = {fundamentalFreq:.2f} Hz, RPM = {estimatedRpm:.2f} ({rpmStatus})")
    
    harmonics = []
    for m in range(1, 6):
        harmonicFreq = m * fundamentalFreq
        if harmonicFreq <= 1000:
            harmonics.append(harmonicFreq)
    
    plt.figure(figsize=(10, 5))
    plt.plot(freqs, mag, label='Spectrum')
    plt.axvline(fundamentalFreq, color='r', linestyle='--', label=f'Fundamental ({fundamentalFreq:.2f} Hz)')
    for i, harmonicFreq in enumerate(harmonics[1:], 2):
        plt.axvline(harmonicFreq, color='g', linestyle=':', label=f'Harmonic {i} ({harmonicFreq:.2f} Hz)')
    plt.title(f'{signalName} DFT (RPM: {estimatedRpm:.2f})')
    plt.xlabel('Frequency (Hz)')
    plt.ylabel('Magnitude')
    plt.xlim(0, 1000)
    plt.legend()
    plt.grid()
    plt.show()
    
    return fundamentalFreq, estimatedRpm

    if fundamentalFreq is not None and 'sig_c' in signalName:
        fundamentalFreqs.append(fundamentalFreq)
        estimatedRpms.append(estimatedRpm)
